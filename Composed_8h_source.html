136// Specifialized macros to generate getters/setters for `UA_*Attribute` classes.
137// The `specifiedAttributes` mask is automatically updated in the setter methods.
138// A fluent interface is used for the setter methods.
139
140// NOLINTNEXTLINE
141#define UAPP_NODEATTR(Type, suffix, member, flag) \
142 UAPP_COMPOSED_GETTER(Type, get##suffix, member) \
143 auto& set##suffix(Type member) noexcept { \
144 handle()->specifiedAttributes |= flag; \
145 handle()->member = member; \
146 return *this; \
147 }
148
149// NOLINTNEXTLINE
150#define UAPP_NODEATTR_CAST(Type, suffix, member, flag) \
151 UAPP_COMPOSED_GETTER_CAST(Type, get##suffix, member) \
152 auto& set##suffix(Type member) noexcept { \
153 handle()->specifiedAttributes |= flag; \
154 handle()->member = static_cast<decltype(handle()->member)>(member); \
155 return *this; \
156 }
157
158// NOLINTNEXTLINE
159#define UAPP_NODEATTR_WRAPPER(WrapperType, suffix, member, flag) \
160 UAPP_COMPOSED_GETTER_WRAPPER_CONST(WrapperType, get##suffix, member) \
161 auto& set##suffix(const WrapperType& member) { \
162 handle()->specifiedAttributes |= flag; \
163 asWrapper<WrapperType>(handle()->member) = member; \
164 return *this; \
165 }
166
167// NOLINTNEXTLINE
168#define UAPP_NODEATTR_ARRAY(Type, suffix, memberArray, memberSize, flag) \
169 UAPP_COMPOSED_GETTER_ARRAY(Type, get##suffix, memberArray, memberSize) \
170 auto& set##suffix(const std::vector<Type>& memberArray) { \
171 handle()->specifiedAttributes |= flag; \
172 UA_Array_delete( \
173 handle()->memberArray, handle()->memberSize, &detail::guessDataType<Type>() \
174 ); \
175 handle()->memberArray = detail::toNativeArrayAlloc( \
176 memberArray.begin(), memberArray.end() \
177 ); \
178 handle()->memberSize = memberArray.size(); \
179 return *this; \
180 }
