112// Specifialized macros to generate getters/setters for `UA_*Attribute` classes.
113// The `specifiedAttributes` mask is automatically updated in the setter methods.
114// A fluent interface is used for the setter methods.
115
116// NOLINTNEXTLINE
117#define UAPP_NODEATTR(Type, suffix, member, flag) \
118 UAPP_COMPOSED_GETTER(Type, get##suffix, member) \
119 auto& set##suffix(Type member) noexcept { \
120 handle()->specifiedAttributes |= flag; \
121 handle()->member = member; \
122 return *this; \
123 }
124
125// NOLINTNEXTLINE
126#define UAPP_NODEATTR_CAST(Type, suffix, member, flag) \
127 UAPP_COMPOSED_GETTER_CAST(Type, get##suffix, member) \
128 auto& set##suffix(Type member) noexcept { \
129 handle()->specifiedAttributes |= flag; \
130 handle()->member = static_cast<decltype(handle()->member)>(member); \
131 return *this; \
132 }
133
134// NOLINTNEXTLINE
135#define UAPP_NODEATTR_WRAPPER(WrapperType, suffix, member, flag) \
136 UAPP_COMPOSED_GETTER_WRAPPER_CONST(WrapperType, get##suffix, member) \
137 auto& set##suffix(const WrapperType& member) { \
138 handle()->specifiedAttributes |= flag; \
139 asWrapper<WrapperType>(handle()->member) = member; \
140 return *this; \
141 }
142
143// NOLINTNEXTLINE
144#define UAPP_NODEATTR_ARRAY(Type, suffix, memberArray, memberSize, flag) \
145 UAPP_COMPOSED_GETTER_ARRAY(Type, get##suffix, memberArray, memberSize) \
146 auto& set##suffix(const std::vector<Type>& memberArray) { \
147 handle()->specifiedAttributes |= flag; \
148 UA_Array_delete( \
149 handle()->memberArray, \
150 handle()->memberSize, \
151 detail::getUaDataType(detail::guessTypeIndex<Type>()) \
152 ); \
153 handle()->memberArray = detail::toNativeArrayAlloc( \
154 memberArray.begin(), memberArray.end() \
155 ); \
156 handle()->memberSize = memberArray.size(); \
157 return *this; \
158 }
