24/**
25 * @defgroup MonitoredItem MonitoredItem service set
26 * Subscribe to data and events.
27 *
28 * Note the difference between Subscriptions and MonitoredItems. Subscriptions are used to report
29 * back notifications. MonitoredItems are used to generate notifications. Every MonitoredItem is
30 * attached to exactly one Subscription. And a Subscription can contain many MonitoredItems.
31 *
32 * MonitoredItems can also be registered locally (server-side). Notifications are then forwarded
33 * to a user-defined callback instead of a remote client.
34 *
35 * @see Subscription
36 * @see https://reference.opcfoundation.org/Core/Part4/v105/docs/5.12
37 * @ingroup Services
38 * @{
39 */

41/**
42 * Monitoring parameters with default values from open62541.
43 * Parameters are passed by reference because illegal parameters can be revised by the server.
44 * The updated parameters reflect the actual values that the server will use.
45 * @see https://reference.opcfoundation.org/Core/Part4/v105/docs/7.21
46 */
47struct MonitoringParameters {
48 /// Timestamps to be transmitted. Won't be revised by the server.
49 TimestampsToReturn timestamps = TimestampsToReturn::Both;
50 /// Interval in milliseconds that defines the fastest rate at which the MonitoredItem should be
51 /// accessed and evaluated. The following values have special meaning:
52 /// - `0.0` to use the fastest practical rate
53 /// - `-1.0` to use the default sampling interval (publishing interval of the subscription)
54 double samplingInterval = 250.0;
55 /// Filter is used by the server to determine if the MonitoredItem should generate
56 /// notifications. The filter parameter type is an extensible parameter type and can be, for
57 /// example, of type DataChangeFilter, EventFilter or AggregateFilter.
58 /// @see https://reference.opcfoundation.org/Core/Part4/v105/docs/7.22
59 ExtensionObject filter;
60 /// Size of the MonitoringItem queue.
61 /// The following values have special meaning:
62 /// - `0` to retrieve the server's default queue size
63 /// - `1` to retrieve the server's minimum queue size
64 /// In the case of a queue overflow, an Event of the type EventQueueOverflowEventType is
65 /// generated.
66 uint32_t queueSize = 1;
67 /// Discard policy when the queue is full.
68 /// - `true`: the oldest (first) notification in the queue is discarded
69 /// - `false`: the last notification added to the queue gets replaced with the new notification
70 bool discardOldest = true;
71};
