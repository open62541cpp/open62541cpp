25/**
26 * @defgroup MonitoredItem MonitoredItem service set
27 * Subscribe to data and events.
28 *
29 * Note the difference between Subscriptions and MonitoredItems. Subscriptions are used to report
30 * back notifications. Monitored items are used to generate notifications. Every monitored item is
31 * attached to exactly one subscription. And a subscription can contain many monitored items.
32 *
33 * Monitored items can also be registered locally (server-side). Notifications are then forwarded
34 * to a user-defined callback instead of a remote client.
35 *
36 * @see Subscription
37 * @see https://reference.opcfoundation.org/Core/Part4/v105/docs/5.12
38 * @ingroup Services
39 * @{
40 */

42/**
43 * Extended monitoring parameters with default values from open62541.
44 * This is an extended version of `UA_MonitoringParameters` with the `timestamps` parameter.
45 * Parameters are passed by reference because illegal parameters can be revised by the server.
46 * The updated parameters reflect the actual values that the server will use.
47 * @see https://reference.opcfoundation.org/Core/Part4/v105/docs/7.21
48 */
49struct MonitoringParametersEx {
50 /// Timestamps to be transmitted. Won't be revised by the server.
51 TimestampsToReturn timestamps = TimestampsToReturn::Both;
52 /// Interval in milliseconds that defines the fastest rate at which the MonitoredItem should be
53 /// accessed and evaluated. The following values have special meaning:
54 /// - `0.0` to use the fastest practical rate
55 /// - `-1.0` to use the default sampling interval (publishing interval of the subscription)
56 double samplingInterval = 250.0;
57 /// Filter is used by the server to determine if the MonitoredItem should generate
58 /// notifications. The filter parameter type is an extensible parameter type and can be, for
59 /// example, of type DataChangeFilter, EventFilter or AggregateFilter.
60 /// @see https://reference.opcfoundation.org/Core/Part4/v105/docs/7.22
61 ExtensionObject filter;
62 /// Size of the MonitoringItem queue.
63 /// The following values have special meaning:
64 /// - `0` to retrieve the server's default queue size
65 /// - `1` to retrieve the server's minimum queue size
66 /// In the case of a queue overflow, an Event of the type EventQueueOverflowEventType is
67 /// generated.
68 uint32_t queueSize = 1;
69 /// Discard policy when the queue is full.
70 /// - `true`: the oldest (first) notification in the queue is discarded
71 /// - `false`: the last notification added to the queue gets replaced with the new notification
72 bool discardOldest = true;
73};
