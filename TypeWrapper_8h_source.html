18 *
19 * Safe wrapper classes for heap-allocated open62541 `UA_*` types to prevent memory leaks.

25/**
26 * Template base class to wrap `UA_*` type objects.
27 *
28 * Zero cost abstraction to wrap the C API objects and delete them on destruction. The derived
29 * classes should implement specific constructors to convert from other data types.
30 *
31 * `TypeWrapper<T, typeIndex>` is pointer-interconvertible to `T`. Use asWrapper(NativeType&) or
32 * asWrapper(constNativeType&) to cast native object references to wrapper object references.
33 *
34 * @warning No virtual constructor defined, don't implement a destructor in the derived classes.
35 * @ingroup TypeWrapper
36 */
37template <typename T, TypeIndex typeIndex>
38class TypeWrapper {
39public:
40 static_assert(typeIndex < UA_TYPES_COUNT);
41
42 using TypeWrapperBase = TypeWrapper<T, typeIndex>;
43 using NativeType = T;

201/**
202 * Cast native `UA_*` type object pointers to TypeWrapper object pointers.
203 * This is especially helpful to avoid copies in getter methods of composed types.
204 * @see https://en.cppreference.com/w/cpp/language/static_cast#pointer-interconvertible
205 * @see https://github.com/open62541pp/open62541pp/issues/30
206 * @ingroup TypeWrapper
207 */
208template <typename WrapperType, typename NativeType = typename WrapperType::NativeType>
209constexpr WrapperType* asWrapper(NativeType* native) noexcept {
210 detail::assertIsWrappedType<WrapperType, NativeType>();
211 detail::assertIsPointerInterconvertible<WrapperType, NativeType>();
212 return static_cast<WrapperType*>(static_cast<void*>(native));
213}
