18 * UA_StatusCode wrapper class.
19 * This type is not derived from TypeWrapper. @ref UA_StatusCode is just an alias for `uint32_t`.
20 * StatusCode can be used interchangeably with @ref UA_StatusCode due to implicit conversions
21 * (without any overhead) but provides some methods to simplify the handling with status codes.
22 * @see statuscodes.h
23 * @see https://reference.opcfoundation.org/Core/Part4/v105/docs/7.39
24 * @ingroup TypeWrapper
25 */
26class StatusCode {
27public:
28 /// Create a StatusCode with the default status code `UA_STATUSCODE_GOOD`.
29 constexpr StatusCode() noexcept = default;
30
31 constexpr StatusCode(UA_StatusCode code) noexcept // NOLINT, implicit wanted
32 : code_(code) {}
33
34 /// Implicit conversion to UA_StatusCode.
35 constexpr operator UA_StatusCode() noexcept { // NOLINT, implicit wanted
36 return code_;
37 }
38
39 /// Explicitly get underlying UA_StatusCode.
40 constexpr UA_StatusCode get() const noexcept {
41 return code_;
42 }
43
44 /// Get human-readable name of the StatusCode.
45 /// This feature might be disabled to create a smaller binary with the
46 /// `UA_ENABLE_STATUSCODE_DESCRIPTIONS` build-flag. Then the function returns an empty string
47 /// for every StatusCode.
48 std::string_view name() const noexcept {
49 return {UA_StatusCode_name(code_)};
50 }
51
52 /// Check if the status code is good.
53 constexpr bool isGood() const noexcept {
54 return detail::isGoodStatus(code_);
55 }
56
57 /// Check if the status code is uncertain.

60 }

64 return detail::isBadStatus(code_);
65 }
66
67 /// Throw a BadStatus exception if the status code is bad.
68 /// @exception BadStatus
69 void throwIfBad() const {
70 detail::throwOnBadStatus(code_);
71 }

73private:
74 UA_StatusCode code_{};
75};
