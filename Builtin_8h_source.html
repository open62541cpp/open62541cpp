26/**
27 * UA_StatusCode wrapper class.
28 * This type is not derived from TypeWrapper. @ref UA_StatusCode is just an alias for `uint32_t`.
29 * StatusCode can be used interchangeably with @ref UA_StatusCode due to implicit conversions
30 * (without any overhead) but provides some methods to simplify the handling with status codes.
31 * @see statuscodes.h
32 * @see https://reference.opcfoundation.org/Core/Part4/v105/docs/7.39
33 * @ingroup TypeWrapper
34 */
35class StatusCode {
36public:
37 using NativeType = UA_StatusCode;
38
39 /// Create a StatusCode with the default status code `UA_STATUSCODE_GOOD`.
40 constexpr StatusCode() noexcept = default;
41
42 constexpr StatusCode(UA_StatusCode code) noexcept // NOLINT, implicit wanted
43 : code_(code) {}
44
45 /// Implicit conversion to UA_StatusCode.
46 constexpr operator UA_StatusCode() const noexcept { // NOLINT, implicit wanted
47 return code_;
48 }
49
50 /// Explicitly get underlying UA_StatusCode.
51 constexpr UA_StatusCode get() const noexcept {
52 return code_;
53 }
54
55 /// Get human-readable name of the StatusCode.
56 /// This feature might be disabled to create a smaller binary with the
57 /// `UA_ENABLE_STATUSCODE_DESCRIPTIONS` build-flag. Then the function returns an empty string
58 /// for every StatusCode.
59 std::string_view name() const noexcept {
60 return {UA_StatusCode_name(code_)};
61 }
62
63 /// Check if the status code is good.
64 constexpr bool isGood() const noexcept {
65 return detail::isGoodStatus(code_);
66 }
67
68 /// Check if the status code is uncertain.
69 constexpr bool isUncertain() const noexcept {
70 return detail::isUncertainStatus(code_);
71 }
72
73 /// Check if the status code is bad.
74 constexpr bool isBad() const noexcept {
75 return detail::isBadStatus(code_);
76 }
77
78 /// Throw a BadStatus exception if the status code is bad.
79 /// @exception BadStatus
80 void throwIfBad() const {
81 detail::throwOnBadStatus(code_);
82 }
83
84private:
85 UA_StatusCode code_{};
86};
