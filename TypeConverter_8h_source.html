41/**
42 * Type conversion from and to native types.
43 *
44 * Native types can be both `UA_*` types and wrapper classes (like `UA_Guid` and `Guid`).
45 * The `TypeConverter` is mainly used within the `Variant` class to set/get non-native types.
46 *
47 * Template specializations can be added for conversions of arbitrary types:
48 * @code
49 * namespace ::opcua {
50 * template <>
51 * struct TypeConverter<MyCustomType> {
52 * using NativeType = Guid;
53 *
54 * static void fromNative(const NativeType& src, MyCustomType& dst) {
55 * // ...
56 * }
57 *
58 * static void toNative(const MyCustomType& src, NativeType& dst) {
59 * // ...
60 * }
61 * };
62 * }
63 * @endcode
64 */
65template <typename T, typename Enable = void>
66struct TypeConverter;
