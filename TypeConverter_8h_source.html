62template <typename T>
63constexpr bool isValidTypeCombination(const UA_DataType* dataType) {
64 for (auto typeIndex : TypeConverter<T>::ValidTypes::toArray()) {
65 // compare pointer
66 if (dataType == &UA_TYPES[typeIndex]) { // NOLINT
67 return true;
68 }
69 // compare type ids
70 if (dataType->typeId == UA_TYPES[typeIndex].typeId) { // NOLINT
71 return true;
72 }
73 }
74 return false;
75}

113/// Create and convert vector from native array.
114template <typename T, typename NativeType = typename TypeConverter<T>::NativeType>
115[[nodiscard]] std::vector<T> fromNativeArray(NativeType* array, size_t size) {
116 if constexpr (isBuiltinType<T>() && std::is_fundamental_v<T>) {
117 return std::vector<T>(array, array + size); // NOLINT
118 } else {
119 std::vector<T> result(size);
120 for (size_t i = 0; i < size; ++i) {
121 TypeConverter<T>::fromNative(array[i], result[i]); // NOLINT
122 }
123 return result;
124 }

170template <typename InputIt>

172 using ValueType = typename std::iterator_traits<InputIt>::value_type;

194template <typename T, TypeIndex... typeIndexes>
195struct TypeConverterNative {
196 using ValueType = T;
197 using NativeType = T;
198 using ValidTypes = TypeIndexList<typeIndexes...>;
199
200 static_assert(ValidTypes::size() >= 1);
201
202 static void fromNative(const T& src, T& dst) {
203 if constexpr (std::is_fundamental_v<T>) {
204 dst = src;
205 } else {
206 // just take first type -> underlying memory layout of all types should be the same
207 constexpr auto typeIndexGuess = ValidTypes::toArray().at(0);
208 // clear first
209 UA_clear(&dst, &getUaDataType<typeIndexGuess>());
210 // deep copy
211 const auto status = UA_copy(&src, &dst, &getUaDataType<typeIndexGuess>());
212 throwOnBadStatus(status);
213 }
214 }
215
216 static void toNative(const T& src, T& dst) {
217 fromNative(src, dst);
218 }
219};

221template <typename T>
222constexpr bool isNativeType() {
223 using ValueType = typename TypeConverter<T>::ValueType;
224 using NativeType = typename TypeConverter<T>::NativeType;
225 return std::is_same_v<ValueType, NativeType>;
226}
