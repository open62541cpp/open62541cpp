204template <typename T, typename Tag>
205template <auto T::*field>
206DataTypeBuilder<T, Tag>& DataTypeBuilder<T, Tag>::addField(
207 const char* fieldName, const UA_DataType& fieldType
208) {
209 using TMember = detail::MemberTypeT<decltype(field)>;
210 static_assert(
211 std::is_same_v<Tag, detail::TagDataTypeStruct>,
212 "Built type must be a struct or class to add members"
213 );
214 assert(sizeof(detail::UnqualifiedT<TMember>) == fieldType.memSize); // NOLINT
215 if (std::is_pointer_v<TMember>) {
216 dataType_.setTypeKind(UA_DATATYPEKIND_OPTSTRUCT);
217 }
218 if (std::is_pointer_v<TMember> || !fieldType.pointerFree) {
219 dataType_.setPointerFree(false);
220 }
221 fields_.push_back({
222 sizeof(TMember),
223 detail::offsetOfMember(field),
224 detail::createDataTypeMember(
225 fieldName,
226 fieldType,
227 {}, // calculate padding between members later
228 false,
229 std::is_pointer_v<TMember>
230 ),
231 });
232 return *this;
233}

235template <typename T, typename Tag>
236template <auto T::*fieldSize, auto T::*fieldArray>
237DataTypeBuilder<T, Tag>& DataTypeBuilder<T, Tag>::addField(
238 const char* fieldName, const UA_DataType& fieldType
239) {
240 using TSize = detail::MemberTypeT<decltype(fieldSize)>;
241 using TArray = detail::MemberTypeT<decltype(fieldArray)>;
242 static_assert(
243 std::is_same_v<Tag, detail::TagDataTypeStruct>,
244 "Built type must be a struct or class to add members"
245 );
246 static_assert(std::is_integral_v<TSize>, "TSize must be an integral type");
247 static_assert(std::is_pointer_v<TArray>, "TArray must be a pointer");
248 // NOLINTNEXTLINE
249 assert(
250 detail::offsetOfMember(fieldArray) == detail::offsetOfMember(fieldSize) + sizeof(TSize) &&
251 "No padding between members size and array allowed"
252 );
253 assert(sizeof(detail::UnqualifiedT<TArray>) == fieldType.memSize); // NOLINT
254 dataType_.setPointerFree(false);
255 fields_.push_back({
256 sizeof(TSize) + sizeof(TArray),
257 detail::offsetOfMember(fieldSize), // offset/padding related to size field
258 detail::createDataTypeMember(
259 fieldName,
260 fieldType,
261 {}, // calculate padding between members later
262 true,
263 false
264 ),
265 });
266 return *this;

269template <typename T, typename Tag>
270template <auto T::*memberUnion, typename TField>
271DataTypeBuilder<T, Tag>& DataTypeBuilder<T, Tag>::addUnionField(
272 const char* fieldName, const UA_DataType& fieldType
273) {
274 using TUnion = detail::MemberTypeT<decltype(memberUnion)>;
275 static_assert(
276 std::is_same_v<Tag, detail::TagDataTypeUnion>,
277 "Built type must be a union to add union fields"
278 );
279 static_assert(std::is_union_v<TUnion>, "TUnion must be a union");
280 static_assert(sizeof(TField) <= sizeof(TUnion), "TField exceeds size of union");
281 const auto offset = detail::offsetOfMember(memberUnion);
282 assert(offset > 0 && "A union type must consist of a switch field and a union"); // NOLINT
283 assert(sizeof(detail::UnqualifiedT<TField>) == fieldType.memSize); // NOLINT
284 if (std::is_pointer_v<TField> || !fieldType.pointerFree) {
285 dataType_.setPointerFree(false);
286 }
287 fields_.push_back({
288 sizeof(TField),
289 offset,
290 detail::createDataTypeMember(
291 fieldName,
292 fieldType,
293 static_cast<uint8_t>(offset), // padding = offset of each field
294 false,
295 std::is_pointer_v<TField>
296 ),
297 });
298 return *this;
299}

301template <typename T, typename Tag>
302[[nodiscard]] DataType DataTypeBuilder<T, Tag>::build() {
303 static_assert(!std::is_same_v<Tag, detail::TagDataTypeAny>);
304 // sort members by offset
305 std::sort(fields_.begin(), fields_.end(), [](const auto& lhs, const auto& rhs) {
306 return lhs.offset < rhs.offset;
307 });
308 // calculate padding of struct members
309 if constexpr (std::is_same_v<Tag, detail::TagDataTypeStruct>) {
310 for (auto it = fields_.begin(); it < fields_.end(); ++it) {
311 if (it == fields_.begin()) {
312 it->dataTypeMember.padding = static_cast<uint8_t>(it->offset);
313 } else {
314 it->dataTypeMember.padding = static_cast<uint8_t>(
315 it->offset - (std::prev(it)->offset + std::prev(it)->memSize)
316 );
317 }
318 }
319 }
320 // generate and set members array
321 std::vector<DataTypeMember> dataTypeMembers(fields_.size());
322 std::transform(fields_.cbegin(), fields_.cend(), dataTypeMembers.begin(), [](auto&& m) {
323 return m.dataTypeMember;
324 });
325 dataType_.setMembers(dataTypeMembers);
326 return dataType_;
327}
