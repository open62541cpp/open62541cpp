404template <typename T>
405T Variant::getScalarCopyImpl() const {
406 if constexpr (detail::isRegisteredType<T>) {
407 return detail::copy(getScalar<T>(), opcua::getDataType<T>());
408 } else {
409 using Native = typename TypeConverter<T>::NativeType;
410 T result{};
411 TypeConverter<T>::fromNative(getScalar<Native>(), result);
412 return result;
413 }
414}
415
416template <typename T>
417std::vector<T> Variant::getArrayCopyImpl() const {
418 std::vector<T> result(handle()->arrayLength);
419 if constexpr (detail::isRegisteredType<T>) {
420 auto native = getArray<T>();
421 std::transform(native.begin(), native.end(), result.begin(), [](auto&& value) {
422 return detail::copy(value, opcua::getDataType<T>());
423 });
424 } else {
425 using Native = typename TypeConverter<T>::NativeType;
426 auto native = getArray<Native>();
427 for (size_t i = 0; i < native.size(); ++i) {
428 TypeConverter<T>::fromNative(native[i], result[i]);
429 }
430 }
431 return result;
432}

475template <typename InputIt>
476void Variant::setArrayCopyImpl(InputIt first, InputIt last, const UA_DataType& dataType) {
477 using ValueType = typename std::iterator_traits<InputIt>::value_type;
478 const size_t size = std::distance(first, last);
479 auto native = detail::allocateArrayUniquePtr<ValueType>(size, dataType);
480 std::transform(first, last, native.get(), [&](const ValueType& value) {
481 return detail::copy(value, dataType);
482 });
483 setArrayImpl(native.release(), size, dataType, UA_VARIANT_DATA); // move ownership
484}
485
486template <typename InputIt>
487void Variant::setArrayCopyConvertImpl(InputIt first, InputIt last) {
488 using ValueType = typename std::iterator_traits<InputIt>::value_type;
489 using Native = typename TypeConverter<ValueType>::NativeType;
490 const auto& dataType = opcua::getDataType<Native>();
491 const size_t size = std::distance(first, last);
492 auto native = detail::allocateArrayUniquePtr<Native>(size, dataType);
493 for (size_t i = 0; i < size; ++i) {
494 TypeConverter<ValueType>::toNative(*first++, native.get()[i]); // NOLINT
495 }
496 setArrayImpl(native.release(), size, dataType, UA_VARIANT_DATA); // move ownership
497}
