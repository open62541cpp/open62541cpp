28/**
29 * High-level node class to access node attribute, browse and populate address space.
30 *
31 * The Node API is just a more convenient way of using the free functions in the `services`
32 * namespace.
33 *
34 * Node objects are useful as-is but they do not expose the entire OPC UA protocol. You can get
35 * access to the associated NodeId instance with the getNodeId() method and apply the native
36 * open62541 functions or the free functions in the `services` namespace.
37 *
38 * @see Services
39 */

269 /// Browse referenced nodes (only local nodes).
270 std::vector<Node> browseReferencedNodes(
271 BrowseDirection browseDirection = BrowseDirection::Both,
272 const NodeId& referenceType = ReferenceTypeId::References,
273 bool includeSubtypes = true,
274 Bitmask<NodeClass> nodeClassMask = NodeClass::Unspecified
275 ) {
276 auto refs = services::browseAll(
277 connection_,
278 BrowseDescription(
279 nodeId_,
280 browseDirection,
281 referenceType,
282 includeSubtypes,
283 nodeClassMask,
284 BrowseResultMask::TargetInfo // only node id required here
285 )
286 );
287 std::vector<Node> nodes;
288 nodes.reserve(refs.size());
289 for (auto&& ref : refs) {
290 if (ref.getNodeId().isLocal()) {
291 nodes.emplace_back(connection_, std::move(ref.getNodeId().getNodeId()));
292 }
293 }
294 return nodes;
295 }

656 Node browseObjectProperty(const QualifiedName& propertyName) {
657 auto result = services::translateBrowsePathToNodeIds(
658 connection_,

660 );
661 result.getStatusCode().throwIfBad();
