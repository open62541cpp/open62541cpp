66template <typename Clock, typename Duration>
67DateTime DateTime::fromTimePoint(std::chrono::time_point<Clock, Duration> timePoint) {
68 static constexpr int64_t dateTimeUnixEpoch = UA_DATETIME_UNIX_EPOCH;
69 const int64_t sinceUnixEpoch =
70 std::chrono::duration_cast<UaDuration>(timePoint.time_since_epoch()).count();
71 return DateTime(dateTimeUnixEpoch + sinceUnixEpoch);
72}

74template <typename Clock, typename Duration>
75std::chrono::time_point<Clock, Duration> DateTime::toTimePoint() const {
76 static constexpr std::chrono::time_point<Clock, Duration> unixEpoch{};
77 const auto dateTime = get();
78 if (dateTime < UA_DATETIME_UNIX_EPOCH) {
79 return unixEpoch;
80 }
81 const auto sinceEpoch = UaDuration(dateTime - UA_DATETIME_UNIX_EPOCH);
82 return unixEpoch + std::chrono::duration_cast<Duration>(sinceEpoch);
83}
