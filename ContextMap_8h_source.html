13/**
14 * Thread-safe map for context objects.
15 * Context objects are reference as `void*` pointers in open62541 functions/callbacks. To prevent
16 * pointer-invalidation, the objects are stored as unique pointers.
17 * Stale objects will be removed when new objects are stored in the map.
18 */
19template <typename Key, typename Item>
20class ContextMap {
21public:
22 /// Access or insert specified element
23 Item* operator[](Key key) {
24 eraseStale();
25 auto lock = acquireLock();
26 auto& item = map_[key];
27 if (item == nullptr) {
28 item = std::make_unique<Item>(); // allocate item if empty
29 }
30 return item.get();
31 }
32
33 /// Inserts an element or assigns to the current element if the key already exists
34 Item* insert(Key key, std::unique_ptr<Item>&& item) {
35 eraseStale();
36 auto lock = acquireLock();
37 return map_.insert_or_assign(key, std::move(item)).first->second.get();
38 }
39
40 size_t erase(Key key) {
41 auto lock = acquireLock();
42 return map_.erase(key);
43 }
44
45 size_t eraseStale() {
46 const size_t count = map_.size();
47 if constexpr (IsStaleable<Item>::value) {
48 auto lock = acquireLock();
49 for (auto it = map_.begin(); it != map_.end();) {
50 if (it->second->stale) {
51 it = map_.erase(it);
52 } else {
53 ++it;
54 }
55 }
56 }
57 return count - map_.size();
58 }
59
60 bool contains(Key key) const {
61 auto lock = acquireLock();
62 return map_.count(key) > 0;
63 }
64
65 const Item* find(Key key) const {
66 auto lock = acquireLock();
67 auto it = map_.find(key);
68 if (it != map_.end()) {
69 return it->second.get();
70 }
71 return nullptr;
72 }
73
74 /// Acquire the lock/mutex for unique access to the underlying map.
75 [[nodiscard]] auto acquireLock() const {
76 return std::unique_lock(mutex_);
77 }
78
79 /// Get access to the underlying map. Use `acquireLock` before any operation.
80 const auto& underlying() const noexcept {
81 assertLocked();
82 return map_;
83 }
84
85private:
86 void assertLocked() const {
87 assert(std::unique_lock(mutex_, std::defer_lock).try_lock() == false);
88 }
89
90 std::map<Key, std::unique_ptr<Item>> map_;
