11/**
12 * Value callbacks for variable nodes.
13 *
14 * Value callbacks allow to synchronize a variable value with an external representation.
15 * The attached callbacks are executed before every read and after every write operation.
16 * @see https://www.open62541.org/doc/1.3/tutorial_server_datasource.html
17 */
18struct ValueCallback {
19 /**
20 * Called before the value attribute is read.
21 *
22 * It is possible to write into the value attribute during onBeforeRead (using e.g.
23 * services::writeValue or Node::writeValue). The node is re-opened afterwards so that changes
24 * are considered in the following read operation.
25 *
26 * @param value Current value before the read operation
27 */
28 std::function<void(const DataValue& value)> onBeforeRead;
29
30 /**
31 * Called after writing the value attribute.
32 *
33 * The node is re-opened after writing so that the new value is visible in the callback.
34 *
35 * @param value New value after the write operation
36 */
37 std::function<void(const DataValue& value)> onAfterWrite;
38};

40/**
41 * Data source backend for variable nodes.
42 *
43 * The server redirects every read and write request to callback functions.
44 * Internally, the data source needs to implement its own memory management.
45 * @see https://www.open62541.org/doc/1.3/tutorial_server_datasource.html
46 */
47struct ValueBackendDataSource {
48 /**
49 * Callback to set the read value, the result status and optionally a source timestamp.
50 *
51 * @note Zero-copy operations are possible. It is not required to return a copy of the actual
52 * content data. You can return a pointer to memory owned by the user. Memory can be reused
53 * between read callbacks of a data source, as the result is already encoded on the network
54 * buffer between each read operation.
55 * To use zero-copy reads, set the value of the Variant (DataValue::getValue) without copying,
56 * e.g. with Variant::setScalar or Variant::setArray.
57 *
58 * @param value The DataValue that is returned to the reader
59 * @param range If not empty, then the data source shall return only a selection of the
60 * (nonscalar) data.
61 * Set `UA_STATUSCODE_BADINDEXRANGEINVALID` in `value` if this does not apply
62 * @param timestamp Set the source timestamp of `value` if `true`
63 * @return StatusCode
64 */
65 std::function<StatusCode(DataValue& value, const NumericRange& range, bool timestamp)> read;
66
67 /**
68 * Callback to write the value into a data source.
69 * This function can be empty if the operation is unsupported.
70 *
71 * @param value The DataValue that has been written by the writer
72 * @param range If not empty, then only this selection of (non-scalar) data should be written
73 * into the data source
74 * @return StatusCode
75 */
76 std::function<StatusCode(const DataValue& value, const NumericRange& range)> write;
77};
