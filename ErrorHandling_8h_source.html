94/**
95 * Invoke a function (with `void` or `UA_StatusCode` return type) and catch exceptions.
96 * This is especially useful for C-API callbacks, that are executed within the open62541 event loop.
97 * If no exception is thrown, the generated status code or `UA_STATUSCODE_GOOD` is returned.
98 * If the exception if of type BadStatus, the underlying status code will be returned.
99 * All other exception types will yield `UA_STATUSCODE_BADINTERNALERROR`.
100 */
101template <typename F, typename... Args>
102[[nodiscard]] UA_StatusCode invokeCatchStatus(F&& fn, Args&&... args) noexcept {
103 using ReturnType = std::invoke_result_t<F, Args&&...>;
104 static_assert(
105 std::is_same_v<ReturnType, void> || std::is_same_v<ReturnType, UA_StatusCode>,
106 "Only return types of type void or UA_StatusCode allowed"
107 );
108 try {
109 if constexpr (std::is_same_v<ReturnType, UA_StatusCode>) {
110 return std::invoke(fn, std::forward<Args>(args)...);
111 } else {
112 std::invoke(fn, std::forward<Args>(args)...);
113 return UA_STATUSCODE_GOOD;
114 }
115 } catch (const BadStatus& e) {
116 return e.code();
117 } catch (...) {
118 return UA_STATUSCODE_BADINTERNALERROR;
119 }
120}
