145void resizeArray(T*& array, size_t& size, size_t newSize, const UA_DataType& type) {
146 if (newSize == size) {
147 return;
148 }
149 T* newArray = allocateArray<T>(newSize, type);

151 if (newSize > size) {
152 std::memset(newArray + size, 0, newSize - size); // NOLINT
153 }
