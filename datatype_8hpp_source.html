192 * Builder to create DataType definitions of custom types.
193 *
194 * The attributes `memSize`, `padding`, `pointerFree`, `isArray` and `isOptional` are automatically
195 * deduced from the types itself.

198class DataTypeBuilder {

200 /**
201 * Build a DataType definition for an enum.

203 * @param typeId NodeId of the type

208 /**
209 * Build a DataType definition for a structure.
210 * A structure may have optional fields (pointers).

218 * Build a DataType definition for an union.
219 * Union type consist of a switch field and the actual union.

227 * Add a structure field.
228 * @tparam field Member pointer, e.g. `&S::value`

236 * Add a structure field (derive DataType from `field`).
237 * @overload

244 /**

246 * Arrays must consists of two fields: its size (of type `size_t`) and the pointer to the data.

356template <typename T, typename Tag, typename U>
357template <auto U::*field>
358auto& DataTypeBuilder<T, Tag, U>::addField(const char* fieldName, const UA_DataType& fieldType) {
359 using TMember = detail::MemberTypeT<decltype(field)>;
360 static_assert(
361 std::is_same_v<Tag, detail::TagDataTypeStruct>,
362 "Built type must be a struct or class to add members"
363 );
364 assert(sizeof(std::remove_pointer_t<TMember>) == fieldType.memSize);
365 if (std::is_pointer_v<TMember>) {
366 dataType_.setTypeKind(UA_DATATYPEKIND_OPTSTRUCT);
367 }
368 if (std::is_pointer_v<TMember> || !fieldType.pointerFree) {
369 dataType_.setPointerFree(false);
370 }
371 fields_.push_back({
372 sizeof(TMember),
373 detail::offsetOfMember(field),
374 detail::createDataTypeMember(
375 fieldName,
376 fieldType,
377 {}, // calculate padding between members later
378 false,
379 std::is_pointer_v<TMember>
380 ),
381 });
382 return *this;
383}

385template <typename T, typename Tag, typename U>
386template <auto U::*fieldSize, auto U::*fieldArray>
387auto& DataTypeBuilder<T, Tag, U>::addField(const char* fieldName, const UA_DataType& fieldType) {
388 using TSize = detail::MemberTypeT<decltype(fieldSize)>;
389 using TArray = detail::MemberTypeT<decltype(fieldArray)>;
390 static_assert(
391 std::is_same_v<Tag, detail::TagDataTypeStruct>,
392 "Built type must be a struct or class to add members"
393 );
394 static_assert(std::is_integral_v<TSize>, "TSize must be an integral type");
395 static_assert(std::is_pointer_v<TArray>, "TArray must be a pointer");
396 assert(
397 detail::offsetOfMember(fieldArray) == detail::offsetOfMember(fieldSize) + sizeof(TSize) &&
398 "No padding between members size and array allowed"
399 );
400 assert(sizeof(std::remove_pointer_t<TArray>) == fieldType.memSize);
401 dataType_.setPointerFree(false);
402 fields_.push_back({
403 sizeof(TSize) + sizeof(TArray),
404 detail::offsetOfMember(fieldSize), // offset/padding related to size field
405 detail::createDataTypeMember(
406 fieldName,
407 fieldType,
408 {}, // calculate padding between members later
409 true,
410 false
411 ),
412 });
413 return *this;
414}

416template <typename T, typename Tag, typename U>
417template <auto U::*memberUnion, typename TField>
418auto& DataTypeBuilder<T, Tag, U>::addUnionField(
419 const char* fieldName, const UA_DataType& fieldType
420) {
421 using TUnion = detail::MemberTypeT<decltype(memberUnion)>;
422 static_assert(
423 std::is_same_v<Tag, detail::TagDataTypeUnion>,
424 "Built type must be a union to add union fields"
425 );
426 static_assert(std::is_union_v<TUnion>, "TUnion must be a union");
427 static_assert(sizeof(TField) <= sizeof(TUnion), "TField exceeds size of union");
428 const auto offset = detail::offsetOfMember(memberUnion);
429 assert(offset > 0 && "A union type must consist of a switch field and a union");
430 assert(sizeof(std::remove_pointer_t<TField>) == fieldType.memSize);
431 if (std::is_pointer_v<TField> || !fieldType.pointerFree) {
432 dataType_.setPointerFree(false);
433 }
434 fields_.push_back({
435 sizeof(TField),
436 offset,
437 detail::createDataTypeMember(
438 fieldName,
439 fieldType,
440 static_cast<uint8_t>(offset), // padding = offset of each field
441 false,
442 std::is_pointer_v<TField>
443 ),
444 });
445 return *this;
446}

448template <typename T, typename Tag, typename U>
449[[nodiscard]] DataType DataTypeBuilder<T, Tag, U>::build() {
450 static_assert(!std::is_same_v<Tag, detail::TagDataTypeAny>);
451 // sort members by offset
452 std::sort(fields_.begin(), fields_.end(), [](const auto& lhs, const auto& rhs) {
453 return lhs.offset < rhs.offset;
454 });
455 // calculate padding of struct members
456 if constexpr (std::is_same_v<Tag, detail::TagDataTypeStruct>) {
457 for (auto it = fields_.begin(); it < fields_.end(); ++it) {
458 if (it == fields_.begin()) {
459 it->dataTypeMember.padding = static_cast<uint8_t>(it->offset);
460 } else {
461 it->dataTypeMember.padding = static_cast<uint8_t>(
462 it->offset - (std::prev(it)->offset + std::prev(it)->memSize)
463 );
464 }
465 }
466 }
467 // generate and set members array
468 std::vector<DataTypeMember> dataTypeMembers(fields_.size());
469 std::transform(fields_.cbegin(), fields_.cend(), dataTypeMembers.begin(), [](auto&& m) {
470 return m.dataTypeMember;
471 });
472 dataType_.setMembers(dataTypeMembers);
473 return dataType_;
474}
